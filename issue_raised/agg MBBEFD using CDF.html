This is a demonstration that using the Cumulative Distribution Function F of the MBBEDF does produce accurate answers in this example. A c parameter approaching 0.000 should produce near total individual losses with the MBBDF distribution.

import aggregate as agg, pandas as pd

import scipy.stats as ss

import numpy as np

import matplotlib.pyplot as plt

def bb(c):
    return np.exp(3.1 - 0.15*c*(1+c))


def bg(c):
    return np.exp((0.78 + 0.12*c)*c)


def G(x, c):
    b = bb(c)
    g = bg(c)
    return np.log(((g - 1) * b + (1 - g * b) * b**x) / (1 - b)) / np.log(g * b)

def F(x, c):
    b = bb(c)
    g = bg(c)

    return np.where(x == 1 , 1 , 1-(1-b)/((g-1)*b**(1-x)+(1-g*b)))

swiss = pd.DataFrame(ans, columns=['c', 'mean', 'cv'])

def beta_ab(m, cv):
    v = (m * cv) ** 2
    sev_a = m * (m * (1 - m) / v - 1)
    sev_b = (1 - m) * (m * (1 - m) / v - 1)
    return sev_a, sev_b


a, b = beta_ab(swiss['mean'], swiss.cv)

swiss['a'] = a

swiss['b'] = b

swiss = swiss.set_index('c')
